{% endfor %}
Stampa il primo elemento della sequenza usando forloop.first:
{% for x in list2 %} {% if forloop.first %} Il primo elemento - {{ x }} {% endif %} {% endfor %}
Stampa l'ultimo elemento della lista usando forloop.last:
{% for x in list2 %} {% if forloop.last %} L'ultimo elemento - {{ x }} {% endif %} {% endfor %}

    {% for key, value in my_dict.items %}
